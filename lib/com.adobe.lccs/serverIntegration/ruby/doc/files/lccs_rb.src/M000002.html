  def getRole(r)
    case r.downcase
      when "none"      ; return RTC::UserRole::NONE
      when "lobby"     ; return RTC::UserRole::LOBBY
      when "viewer"    ; return RTC::UserRole::VIEWER
      when "publisher" ; return RTC::UserRole::PUBLISHER
      when "owner"     ; return RTC::UserRole::OWNER
      else
        return r.to_i
    end
  end
