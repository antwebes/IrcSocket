    def getAuthenticationToken(accountSecret, name, id, role)
      token = "x:#{name}::#{@account}:#{id}:#{@room}:#{role.to_i}"
      signed = "#{token}:#{sign(accountSecret, token)}"

      # unencoded
      #return "ext=#{signed}"

      # encoded
      enc = Base64.encode64(signed).gsub("\n","")
      return "exx=#{enc}"
    end
